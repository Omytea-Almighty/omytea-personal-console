# Pinned to 1.56.0 on purpose — do NOT bump without re-testing st.login().
# Streamlit 1.57.0 has a regression (streamlit/streamlit#14991): the
# Tornado->Starlette server swap moved the OAuth `state` into a module-level
# in-memory cache that is lost whenever the cloud container restarts between
# the login redirect and /oauth2callback, producing a hard
# `authlib...MismatchingStateError` on every sign-in. 1.56.0 is the last
# release before that rewrite and is confirmed working. Revisit once 1.58+
# ships the fix.
streamlit==1.56.0
pydantic>=2.5,<3.0
numpy>=1.24

# Native OIDC sign-in — st.login() / st.user (Google account login).
# Streamlit's auth needs Authlib at runtime; it is NOT pulled in
# automatically, so pin it explicitly. Without [auth] secrets the
# console still runs — the account area degrades to a "set up sign-in"
# hint rather than crashing.
Authlib>=1.3.2

# LLM backends — install only the ones you actually use.
# All four are optional; the package degrades gracefully if any
# specific provider's SDK isn't installed. MockBackend works with
# nothing installed.

# Primary free-tier default (Google AI Studio):
google-genai>=0.3,<2.0

# Free-tier fallback (Groq Cloud Llama):
groq>=0.11,<2.0

# Optional power-user backend (user-supplied Anthropic key):
anthropic>=0.40,<2.0

# Optional power-user backend (user-supplied OpenAI key):
openai>=1.30,<2.0

# Optional Cloudflare Workers AI / Ollama backends — both use plain
# HTTP via the `requests` library; no vendor SDK required.
requests>=2.28

# Video ingestion (Mode 5 "Video query") — uses substrate's perception
# layer which requires OpenCV. headless variant avoids GUI deps on
# Linux servers + macOS app bundles.
opencv-python-headless>=4.8,<5.0

# Omytea quantum substrate — now ships as an independent PyPI package
# (as of v0.4.0). No more vendor copy in the dist tree; just `pip install`.
# Set OMYTEA_CONSOLE_MOCK=1 if you want to run without substrate at all.
omytea-quantum-substrate>=0.1.1,<1.0

# Mode 6 "Live webcam" — streamlit-webrtc only. We deliberately DO NOT
# pin `av` here because PyAV needs ffmpeg + pkg-config on the build
# host, which broke the Streamlit Cloud build in v0.4.1 (rolled back).
# streamlit-webrtc itself declares av as a transitive dep; pip + the
# cloud build environment will resolve a wheel-only version (manylinux
# binary wheels exist for Linux Python 3.11-3.13). If streamlit-webrtc
# fails to install, the Mode 6 install-hint fallback engages cleanly.
streamlit-webrtc>=0.47,<1.0

# Iter #44 — durable persistence on Streamlit Cloud via Turso
# (hosted libSQL, 100% SQLite-compatible). Optional dependency:
# the storage layer gracefully falls back to local SQLite if the
# wheel doesn't install or the env vars OMYTEA_TURSO_URL +
# OMYTEA_TURSO_AUTH_TOKEN aren't set. See SETUP_TURSO.md for
# the 3-minute founder setup. Streamlit Cloud's filesystem is
# ephemeral, so without Turso the calibration loop "predict →
# wait N months → score" cannot survive a redeploy.
libsql-experimental>=0.0.30
